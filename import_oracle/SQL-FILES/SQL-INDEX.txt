DECLARE 
    CURSOR 
        C_INDICE 
    IS 
        SELECT INDEX_NAME FROM DBA_INDEXES WHERE NOT INDEX_NAME LIKE 'BIN%' AND INDEX_TYPE <> 'LOB' AND OWNER IN (SELECT sys_context( 'userenv', 'current_schema' ) FROM DUAL) ORDER BY INDEX_NAME;
    L_RECRIAR VARCHAR2(500);
BEGIN
    OPEN C_INDICE;
    LOOP 
        FETCH C_INDICE INTO L_RECRIAR;
        IF NOT C_INDICE%NOTFOUND THEN
            EXECUTE IMMEDIATE 'ALTER INDEX ' || TO_CHAR(L_RECRIAR) || ' REBUILD';
        END IF;
        EXIT WHEN C_INDICE%NOTFOUND;
    END LOOP;
    CLOSE C_INDICE;
END;
/
EXIT
